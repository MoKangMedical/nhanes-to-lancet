Survey Design Setup
# Load required packages
library(survey)
library(dplyr)

# Load NHANES data
data <- read.csv("analysis_data.csv")

# Declare survey design
design <- svydesign(
  id = ~SDMVPSU,
  strata = ~SDMVSTRA,
  weights = ~WTMEC2YR_ADJ,
  data = data,
  nest = TRUE
)

# Weighted logistic regression
model <- svyglm(
  {{ project.outcome }} ~ {{ project.exposure }} + RIDAGEYR + RIAGENDR,
  design = design,
  family = quasibinomial
)

# Results
summary(model)
exp(coef(model))  # Odds ratios
